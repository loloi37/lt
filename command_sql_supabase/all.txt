-- Create memorials table
CREATE TABLE memorials (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
  
  -- Memorial data (all the wizard steps stored as JSON)
  step1 JSONB NOT NULL DEFAULT '{}',
  step2 JSONB NOT NULL DEFAULT '{}',
  step3 JSONB NOT NULL DEFAULT '{}',
  step4 JSONB NOT NULL DEFAULT '{}',
  step5 JSONB NOT NULL DEFAULT '{}',
  step6 JSONB NOT NULL DEFAULT '{}',
  step7 JSONB NOT NULL DEFAULT '{}',
  step8 JSONB NOT NULL DEFAULT '{}',
  
  -- Status and metadata
  status TEXT NOT NULL DEFAULT 'draft', -- 'draft' or 'published'
  slug TEXT UNIQUE, -- URL-friendly identifier (e.g., 'eleanor-thompson-1942')
  
  -- Quick access fields (denormalized for faster queries)
  full_name TEXT,
  birth_date DATE,
  death_date DATE,
  profile_photo_url TEXT,
  cover_photo_url TEXT
);

-- Create index for faster queries
CREATE INDEX idx_memorials_status ON memorials(status);
CREATE INDEX idx_memorials_created_at ON memorials(created_at DESC);
CREATE INDEX idx_memorials_slug ON memorials(slug);

-- Create trigger to auto-update updated_at
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = timezone('utc'::text, now());
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER update_memorials_updated_at 
    BEFORE UPDATE ON memorials
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

-- Enable Row Level Security (RLS) but allow all access for now
ALTER TABLE memorials ENABLE ROW LEVEL SECURITY;

-- Create policy to allow all operations (since no auth for now)
CREATE POLICY "Allow all access to memorials" 
    ON memorials 
    FOR ALL 
    USING (true) 
    WITH CHECK (true);

-- 1. First, create a function that prevents users from updating the 'paid' column
CREATE OR REPLACE FUNCTION check_paid_status_integrity()
RETURNS TRIGGER AS $$
BEGIN
  -- If the user tries to change 'paid' from false to true manually
  IF (OLD.paid = false AND NEW.paid = true) THEN
    -- Check if the request is coming from a 'service_role' (your server)
    -- If it's a normal user, REJECT the change
    IF current_setting('role') != 'service_role' THEN
      RAISE EXCEPTION 'You cannot manually set your archive as paid.';
    END IF;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 2. Attach this trigger to your memorials table
CREATE TRIGGER secure_paid_column
BEFORE UPDATE ON memorials
FOR EACH ROW
EXECUTE FUNCTION check_paid_status_integrity();

-- =============================================
-- TABLE: memorial_versions
-- Stores a snapshot every time a major change happens
-- =============================================




CREATE TABLE IF NOT EXISTS memorial_versions (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    
    -- Which memorial this version belongs to
    memorial_id UUID NOT NULL REFERENCES memorials(id) ON DELETE CASCADE,
    
    -- Version metadata
    version_number INTEGER NOT NULL DEFAULT 1,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    
    -- Who made the change (user ID from Supabase Auth, or null for system)
    created_by UUID REFERENCES auth.users(id) ON DELETE SET NULL,
    created_by_name TEXT,  -- Display name at time of change (denormalized)
    
    -- What changed
    change_summary TEXT NOT NULL,           -- Auto-generated: "Updated Biography", "Added 3 photos"
    change_reason TEXT,                     -- Optional user-provided reason
    change_type TEXT NOT NULL DEFAULT 'manual',  -- 'manual', 'auto_save', 'witness_contribution', 'restore'
    
    -- Which steps were modified
    steps_modified INTEGER[] NOT NULL DEFAULT '{}',  -- e.g. {6, 8} = biography + photos changed
    
    -- Snapshot of the data at this point in time
    -- We store only the steps that changed (diff approach to save space)
    snapshot_data JSONB NOT NULL,           -- Full or partial snapshot of memorial data
    is_full_snapshot BOOLEAN NOT NULL DEFAULT false,  -- true = complete data, false = only changed steps
    
    -- For restore functionality
    is_restored_from UUID REFERENCES memorial_versions(id),  -- If this version was created by restoring another
    
    CONSTRAINT unique_version_per_memorial UNIQUE (memorial_id, version_number)
);

-- =============================================
-- FUNCTION: Get next version number
-- =============================================

CREATE OR REPLACE FUNCTION get_next_version_number(p_memorial_id UUID)
RETURNS INTEGER AS $$
    SELECT COALESCE(MAX(version_number), 0) + 1
    FROM memorial_versions
    WHERE memorial_id = p_memorial_id;
$$ LANGUAGE sql STABLE;

CREATE OR REPLACE FUNCTION enforce_personal_plan_limit()
RETURNS TRIGGER AS $$
DECLARE
    active_paid_count INT;
    user_plan_mode TEXT;
BEGIN
    IF TG_OP = 'INSERT' THEN
        SELECT COUNT(*), MAX(mode) INTO active_paid_count, user_plan_mode
        FROM memorials
        WHERE user_id = NEW.user_id AND paid = true AND deleted = false;

        IF active_paid_count >= 1 AND user_plan_mode = 'personal' THEN
            RAISE EXCEPTION 'Personal plan is limited to 1 archive. Please upgrade to Family.';
        END IF;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_enforce_personal_limit
BEFORE INSERT ON memorials
FOR EACH ROW
EXECUTE FUNCTION enforce_personal_plan_limit();
